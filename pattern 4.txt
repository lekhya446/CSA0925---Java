import java.util.*;
class Star7
{
    public static void main(String args[])
    {
     int i,j,n;
     Scanner sc=new Scanner(System.in);
     System.out.println("enter num");
     n=sc.nextInt();
     for(i=1;i<=n;i++)
     {
      for(j=1;j<=n-i;j++)
     {
      System.out.print("  ");
     }
     int c=1;
     for(j=1;j<=i;j++)
     {
     System.out.print(" "+c+" ");
      c=c*(i-j)/(j+1);
     }
     System.out.println();
     }
   }
}
