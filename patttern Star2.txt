import java.util.*;
class Star2
{  
    public static void main(String args[])
    { 
       int i,j,n;
       Scanner sc=new Scanner(System.in);
       System.out.println("enter num");
        n=sc.nextInt(); 
       for(i=1;i<=n;i++)
       {
        for(j=n-i;j>=0;j--)
       {
        System.out.print("  ");
       }
        for(j=1;j<=i;j++)
       {
        System.out.print(i+" ");
       }
       System.out.println();
       }
    }
}
